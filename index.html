// This script serves prayer requests from a Google Sheet as a JSON API. const SHEET_NAME = 'PrayerRequests'; // Make sure this matches your Google Sheet's name exactly. /** * Handles GET requests to the deployed web app URL. * When a web app URL is hit, this function retrieves all data from the specified sheet * and returns it as a JSON array. */ function doGet(e) { const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); const sheet = spreadsheet.getSheetByName(SHEET_NAME); if (!sheet) { // Log an error if the sheet is not found Logger.log(`Sheet with name "${SHEET_NAME}" not found.`); return ContentService.createTextOutput(JSON.stringify({ error: `Sheet "${SHEET_NAME}" not found.` })) .setMimeType(ContentService.MimeType.JSON); } // Get all data from the sheet, excluding the header row const data = sheet.getDataRange().getValues(); const headers = data.shift(); // Remove the header row and store it const prayerRequests = []; // Convert each row into an object using headers as keys data.forEach(row => { const rowObject = {}; headers.forEach((header, index) => { rowObject[header] = row[index]; }); prayerRequests.push(rowObject); }); // Set CORS headers to allow requests from any domain (your GitHub Pages site) // This is crucial for your GitHub Pages site to be able to fetch the data. const output = ContentService.createTextOutput(JSON.stringify(prayerRequests)); output.setMimeType(ContentService.MimeType.JSON); output.setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); output.setHeader('Access-Control-Allow-Origin', '*'); // Allow all origins for simplicity, but consider limiting if privacy is a concern return output; } /** * A helper function for initial setup to ensure the sheet has headers. * This is not called by the web app, but can be run manually in the Apps Script editor. */ function setupPrayerSheet() { const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); let sheet = spreadsheet.getSheetByName(SHEET_NAME); if (!sheet) { sheet = spreadsheet.insertSheet(SHEET_NAME); sheet.appendRow(['Date', 'Time', 'Name', 'Request']); Logger.log(`Created new sheet: ${SHEET_NAME} with headers.`); } else { // Check if headers exist, if not, add them const existingHeaders = sheet.getRange("A1:D1").getValues()[0]; if (existingHeaders[0] !== 'Date' || existingHeaders[1] !== 'Time' || existingHeaders[2] !== 'Name' || existingHeaders[3] !== 'Request') { sheet.insertRowBefore(1); // Insert a new row at the top sheet.getRange("A1:D1").setValues([['Date', 'Time', 'Name', 'Request']]); Logger.log(`Added headers to existing sheet: ${SHEET_NAME}.`); } else { Logger.log(`Sheet "${SHEET_NAME}" already exists and has correct headers.`); } } }
